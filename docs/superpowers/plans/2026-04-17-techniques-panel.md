```typescript
export interface TransferTaxInput {
  sourceCategory: "taxable" | "cash" | "retirement" | "real_estate" | "business" | "life_insurance";
  sourceSubType: string;
  targetCategory: "taxable" | "cash" | "retirement" | "real_estate" | "business" | "life_insurance";
  targetSubType: string;
  amount: number;
  sourceAccountValue: number;
  sourceAccountBasis: number;
  /** Total after-tax basis across ALL traditional IRAs (for pro-rata rule). */
  allTraditionalIraBasis: number;
  /** Total balance across ALL traditional IRAs (for pro-rata rule). */
  allTraditionalIraBalance: number;
  ownerAge: number;
  /** Roth IRA contribution basis (for penalty-free withdrawal). */
  rothBasis: number;
}

export interface TransferTaxResult {
  taxableOrdinaryIncome: number;
  capitalGain: number;
  earlyWithdrawalPenalty: number;
  label:
    | "tax_free_rollover"
    | "roth_conversion"
    | "taxable_distribution"
    | "early_distribution"
    | "taxable_liquidation";
}

const RETIREMENT_SUB_TYPES = new Set([
  "traditional_ira",
  "401k",
  "roth_ira",
  "roth_401k",
  "529",
]);

const ROTH_SUB_TYPES = new Set(["roth_ira", "roth_401k"]);

const TAX_DEFERRED_SUB_TYPES = new Set(["traditional_ira", "401k"]);

function isRetirement(category: string, subType: string): boolean {
  return category === "retirement" && RETIREMENT_SUB_TYPES.has(subType);
}

function isRoth(subType: string): boolean {
  return ROTH_SUB_TYPES.has(subType);
}

function isTaxDeferred(subType: string): boolean {
  return TAX_DEFERRED_SUB_TYPES.has(subType);
}

export function classifyTransferTax(input: TransferTaxInput): TransferTaxResult {
  const {
    sourceCategory,
    sourceSubType,
    targetCategory,
    targetSubType,
    amount,
    sourceAccountValue,
    sourceAccountBasis,
    allTraditionalIraBasis,
    allTraditionalIraBalance,
    ownerAge,
    rothBasis,
  } = input;

  const srcRetirement = isRetirement(sourceCategory, sourceSubType);
  const tgtRetirement = isRetirement(targetCategory, targetSubType);

  // ── Retirement → Retirement ───────────────────────────────────────────────

  if (srcRetirement && tgtRetirement) {
    // Roth → Roth: always tax-free
    if (isRoth(sourceSubType) && isRoth(targetSubType)) {
      return { taxableOrdinaryIncome: 0, capitalGain: 0, earlyWithdrawalPenalty: 0, label: "tax_free_rollover" };
    }

    // Tax-deferred → Roth: Roth conversion (taxable as ordinary income, no penalty)
    if (isTaxDeferred(sourceSubType) && isRoth(targetSubType)) {
      let taxableAmount = amount;

      // Pro-rata rule: if traditional IRA has basis, only non-basis portion is taxable
      if (allTraditionalIraBasis > 0 && allTraditionalIraBalance > 0) {
        const basisRatio = allTraditionalIraBasis / allTraditionalIraBalance;
        taxableAmount = amount * (1 - basisRatio);
      }

      return {
        taxableOrdinaryIncome: taxableAmount,
        capitalGain: 0,
        earlyWithdrawalPenalty: 0,
        label: "roth_conversion",
      };
    }

    // Tax-deferred → Tax-deferred (IRA → IRA, 401k → IRA): tax-free rollover
    if (isTaxDeferred(sourceSubType) && isTaxDeferred(targetSubType)) {
      return { taxableOrdinaryIncome: 0, capitalGain: 0, earlyWithdrawalPenalty: 0, label: "tax_free_rollover" };
    }

    // Roth 401k → Roth IRA: tax-free rollover
    if (isRoth(sourceSubType) && isRoth(targetSubType)) {
      return { taxableOrdinaryIncome: 0, capitalGain: 0, earlyWithdrawalPenalty: 0, label: "tax_free_rollover" };
    }

    // Default retirement → retirement: tax-free rollover
    return { taxableOrdinaryIncome: 0, capitalGain: 0, earlyWithdrawalPenalty: 0, label: "tax_free_rollover" };
  }

  // ── Retirement → Non-Retirement (distribution) ────────────────────────────

  if (srcRetirement && !tgtRetirement) {
    const isEarly = ownerAge < 59.5;

    // Roth distribution: contributions (basis) come out first, tax/penalty-free
    if (isRoth(sourceSubType)) {
      const earningsWithdrawn = Math.max(0, amount - rothBasis);
      const taxable = earningsWithdrawn;
      const penalty = isEarly ? earningsWithdrawn * 0.1 : 0;

      return {
        taxableOrdinaryIncome: taxable,
        capitalGain: 0,
        earlyWithdrawalPenalty: penalty,
        label: isEarly && penalty > 0 ? "early_distribution" : "taxable_distribution",
      };
    }

    // Tax-deferred distribution: fully taxable as ordinary income
    const penalty = isEarly ? amount * 0.1 : 0;
    return {
      taxableOrdinaryIncome: amount,
      capitalGain: 0,
      earlyWithdrawalPenalty: penalty,
      label: isEarly ? "early_distribution" : "taxable_distribution",
    };
  }

  // ── Taxable / Cash → Any (liquidation) ────────────────────────────────────

  if (sourceCategory === "taxable" || sourceCategory === "cash") {
    let capitalGain = 0;
    if (sourceAccountValue > 0 && sourceAccountBasis < sourceAccountValue) {
      const gainRatio = (sourceAccountValue - sourceAccountBasis) / sourceAccountValue;
      capitalGain = amount * gainRatio;
    }
    return {
      taxableOrdinaryIncome: 0,
      capitalGain,
      earlyWithdrawalPenalty: 0,
      label: "taxable_liquidation",
    };
  }

  // ── All other sources (real estate, business, life insurance) ─────────────
  // Treated as liquidation with capital gains on appreciation
  let capitalGain = 0;
  if (sourceAccountValue > 0 && sourceAccountBasis < sourceAccountValue) {
    const gainRatio = (sourceAccountValue - sourceAccountBasis) / sourceAccountValue;
    capitalGain = amount * gainRatio;
  }
  return {
    taxableOrdinaryIncome: 0,
    capitalGain,
    earlyWithdrawalPenalty: 0,
    label: "taxable_liquidation",
  };
}
```

```typescript
import type { Account, Transfer, AccountLedger } from "./types";
import { classifyTransferTax } from "./tax-classification";

export interface TransfersInput {
  transfers: Transfer[];
  accounts: Account[];
  accountBalances: Record<string, number>;
  basisMap: Record<string, number>;
  accountLedgers: Record<string, AccountLedger>;
  year: number;
  ownerAges: { client: number; spouse?: number };
}

export interface TransfersResult {
  taxableOrdinaryIncome: number;
  capitalGains: number;
  earlyWithdrawalPenalty: number;
  byTransfer: Record<string, { amount: number; label: string }>;
}

function getTransferAmount(transfer: Transfer, year: number): number | null {
  if (transfer.mode === "one_time") {
    return year === transfer.startYear ? transfer.amount : null;
  }

  if (transfer.mode === "recurring") {
    if (year < transfer.startYear || (transfer.endYear != null && year > transfer.endYear)) {
      return null;
    }
    return transfer.amount * Math.pow(1 + transfer.growthRate, year - transfer.startYear);
  }

  if (transfer.mode === "scheduled") {
    if (year < transfer.startYear || (transfer.endYear != null && year > transfer.endYear)) {
      return null;
    }
    const scheduleEntry = transfer.schedules.find((s) => s.year === year);
    return scheduleEntry ? scheduleEntry.amount : null;
  }

  return null;
}

export function applyTransfers(input: TransfersInput): TransfersResult {
  const { transfers, accounts, accountBalances, basisMap, accountLedgers, year, ownerAges } = input;

  const result: TransfersResult = {
    taxableOrdinaryIncome: 0,
    capitalGains: 0,
    earlyWithdrawalPenalty: 0,
    byTransfer: {},
  };

  const accountMap = new Map(accounts.map((a) => [a.id, a]));

  // Compute total traditional IRA balance and basis for pro-rata rule
  let allTraditionalIraBalance = 0;
  let allTraditionalIraBasis = 0;
  for (const acct of accounts) {
    if (acct.category === "retirement" && (acct.subType === "traditional_ira" || acct.subType === "401k")) {
      allTraditionalIraBalance += accountBalances[acct.id] ?? 0;
      allTraditionalIraBasis += basisMap[acct.id] ?? 0;
    }
  }

  for (const transfer of transfers) {
    const amount = getTransferAmount(transfer, year);
    if (amount == null || amount <= 0) continue;

    const sourceAcct = accountMap.get(transfer.sourceAccountId);
    const targetAcct = accountMap.get(transfer.targetAccountId);
    if (!sourceAcct || !targetAcct) continue;

    const sourceBalance = accountBalances[transfer.sourceAccountId] ?? 0;
    if (sourceBalance <= 0) continue;

    // Cap at available balance
    const actualAmount = Math.min(amount, sourceBalance);

    const ownerAge =
      sourceAcct.owner === "spouse" ? (ownerAges.spouse ?? ownerAges.client) : ownerAges.client;

    const taxResult = classifyTransferTax({
      sourceCategory: sourceAcct.category,
      sourceSubType: sourceAcct.subType,
      targetCategory: targetAcct.category,
      targetSubType: targetAcct.subType,
      amount: actualAmount,
      sourceAccountValue: sourceBalance,
      sourceAccountBasis: basisMap[transfer.sourceAccountId] ?? 0,
      allTraditionalIraBasis,
      allTraditionalIraBalance,
      ownerAge,
      rothBasis: basisMap[transfer.sourceAccountId] ?? 0,
    });

    // Update balances
    accountBalances[transfer.sourceAccountId] = sourceBalance - actualAmount;
    accountBalances[transfer.targetAccountId] = (accountBalances[transfer.targetAccountId] ?? 0) + actualAmount;

    // Update basis: proportional basis moves with the transfer
    const sourceBasis = basisMap[transfer.sourceAccountId] ?? 0;
    const basisTransferred = sourceBalance > 0 ? sourceBasis * (actualAmount / sourceBalance) : 0;
    basisMap[transfer.sourceAccountId] = sourceBasis - basisTransferred;
    basisMap[transfer.targetAccountId] = (basisMap[transfer.targetAccountId] ?? 0) + basisTransferred;

    // Update ledgers
    const srcLedger = accountLedgers[transfer.sourceAccountId];
    if (srcLedger) {
      srcLedger.distributions += actualAmount;
      srcLedger.endingValue = accountBalances[transfer.sourceAccountId];
      srcLedger.entries.push({
        category: "withdrawal",
        label: `Transfer to ${targetAcct.name}`,
        amount: -actualAmount,
        sourceId: transfer.id,
      });
    }
    const tgtLedger = accountLedgers[transfer.targetAccountId];
    if (tgtLedger) {
      tgtLedger.contributions += actualAmount;
      tgtLedger.endingValue = accountBalances[transfer.targetAccountId];
      tgtLedger.entries.push({
        category: "savings_contribution",
        label: `Transfer from ${sourceAcct.name}`,
        amount: actualAmount,
        sourceId: transfer.id,
      });
    }

    // Accumulate tax impact
    result.taxableOrdinaryIncome += taxResult.taxableOrdinaryIncome;
    result.capitalGains += taxResult.capitalGain;
    result.earlyWithdrawalPenalty += taxResult.earlyWithdrawalPenalty;
    result.byTransfer[transfer.id] = { amount: actualAmount, label: taxResult.label };
  }

  return result;
}
```

```typescript
import type { Account, Liability, AssetTransaction, AccountLedger, ExtraPayment } from "./types";

// ── Asset Sales ─────────────────────────────────────────────────────────────

export interface AssetSalesInput {
  sales: AssetTransaction[];
  accounts: Account[];
  liabilities: Liability[];
  accountBalances: Record<string, number>;
  basisMap: Record<string, number>;
  accountLedgers: Record<string, AccountLedger>;
  year: number;
  defaultCheckingId: string | undefined;
}

export interface AssetSalesResult {
  capitalGains: number;
  removedAccountIds: string[];
  removedLiabilityIds: string[];
  bySale: Record<string, { proceeds: number; gain: number }>;
}

export function applyAssetSales(input: AssetSalesInput): AssetSalesResult {
  const {
    sales,
    liabilities,
    accountBalances,
    basisMap,
    accountLedgers,
    year,
    defaultCheckingId,
  } = input;

  const result: AssetSalesResult = {
    capitalGains: 0,
    removedAccountIds: [],
    removedLiabilityIds: [],
    bySale: {},
  };

  for (const sale of sales) {
    if (sale.type !== "sell" || sale.year !== year || !sale.accountId) continue;

    const currentValue = accountBalances[sale.accountId] ?? 0;
    const currentBasis = basisMap[sale.accountId] ?? 0;

    const saleValue = sale.overrideSaleValue ?? currentValue;
    const saleBasis = sale.overrideBasis ?? currentBasis;

    // Capital gains
    const gain = Math.max(0, saleValue - saleBasis);

    // Transaction costs
    const costPct = sale.transactionCostPct ?? 0;
    const costFlat = sale.transactionCostFlat ?? 0;
    const transactionCosts = saleValue * costPct + costFlat;

    // Find and pay off linked mortgage
    let mortgagePayoff = 0;
    const linkedMortgage = liabilities.find((l) => l.linkedPropertyId === sale.accountId);
    if (linkedMortgage) {
      mortgagePayoff = linkedMortgage.balance;
      result.removedLiabilityIds.push(linkedMortgage.id);
    }

    // Net proceeds
    const netProceeds = saleValue - transactionCosts - mortgagePayoff;

    // Route proceeds
    const destinationId = sale.proceedsAccountId ?? defaultCheckingId;
    if (destinationId) {
      accountBalances[destinationId] = (accountBalances[destinationId] ?? 0) + netProceeds;

      const destLedger = accountLedgers[destinationId];
      if (destLedger) {
        destLedger.contributions += netProceeds;
        destLedger.endingValue = accountBalances[destinationId];
        destLedger.entries.push({
          category: "income",
          label: `Proceeds from sale of ${sale.name}`,
          amount: netProceeds,
          sourceId: sale.id,
        });
      }
    }

    // Remove sold account
    accountBalances[sale.accountId] = 0;
    basisMap[sale.accountId] = 0;
    result.removedAccountIds.push(sale.accountId);

    const srcLedger = accountLedgers[sale.accountId];
    if (srcLedger) {
      srcLedger.distributions += saleValue;
      srcLedger.endingValue = 0;
      srcLedger.entries.push({
        category: "withdrawal",
        label: `Sold: ${sale.name}`,
        amount: -saleValue,
        sourceId: sale.id,
      });
    }

    result.capitalGains += gain;
    result.bySale[sale.id] = { proceeds: netProceeds, gain };
  }

  return result;
}
```

```typescript
export interface WithdrawalPenaltyInput {
  amount: number;
  accountCategory: string;
  accountSubType: string;
  ownerAge: number;
  rothBasis: number;
}

export function computeWithdrawalPenalty(input: WithdrawalPenaltyInput): number {
  const { amount, accountCategory, accountSubType, ownerAge, rothBasis } = input;

  // Only retirement accounts incur early withdrawal penalty
  if (accountCategory !== "retirement") return 0;

  // No penalty at or after 59.5
  if (ownerAge >= 59.5) return 0;

  // Roth: penalty only on earnings above basis
  if (accountSubType === "roth_ira" || accountSubType === "roth_401k") {
    const earningsWithdrawn = Math.max(0, amount - rothBasis);
    return earningsWithdrawn * 0.1;
  }

  // Tax-deferred: penalty on full amount
  return amount * 0.1;
}
```

```typescript
export async function GET(
  _request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const firmId = await getOrgId();
    const { id } = await params;

    const scenarioId = await getBaseCaseScenarioId(id, firmId);
    if (!scenarioId) {
      return NextResponse.json({ error: "Client not found" }, { status: 404 });
    }

    const rows = await db
      .select()
      .from(transfers)
      .where(and(eq(transfers.clientId, id), eq(transfers.scenarioId, scenarioId)));

    // Load schedules for each transfer
    const transferIds = rows.map((r) => r.id);
    const scheduleRows =
      transferIds.length > 0
        ? await db
            .select()
            .from(transferSchedules)
            .where(
              transferIds.length === 1
                ? eq(transferSchedules.transferId, transferIds[0])
                : undefined as any // Will need inArray for multiple
            )
        : [];

    const schedulesByTransfer = new Map<string, typeof scheduleRows>();
    for (const s of scheduleRows) {
      const list = schedulesByTransfer.get(s.transferId) ?? [];
      list.push(s);
      schedulesByTransfer.set(s.transferId, list);
    }

    const result = rows.map((r) => ({
      ...r,

    const [updated] = await db
      .update(transfers)
      .set(values)
      .where(and(eq(transfers.id, transferId), eq(transfers.clientId, id)))
      .returning();

    // Replace schedules if provided
    if (schedules && Array.isArray(schedules)) {
      await db.delete(transferSchedules).where(eq(transferSchedules.transferId, transferId));
      if (schedules.length > 0) {
        await db.insert(transferSchedules).values(
          schedules.map((s: { year: number; amount: number }) => ({
            transferId,
            year: s.year,
            amount: String(s.amount),
          }))

    }
```

   - **Assets** grouped by category (Cash, Taxable, Retirement, Real Estate, Business, Life Insurance)
   - Each account shows name and ending value for that year
   - Category subtotals
   - **Liabilities** section with each liability's remaining balance
   - **Net Worth** = total assets - total liabilities

The data comes from `projectionYear.accountLedgers` (ending values) and `projectionYear.portfolioAssets` (category totals).

```typescript
"use client";

import { useEffect, useState } from "react";
import { runProjection } from "@/engine/projection";
import type { ProjectionYear } from "@/engine/types";

interface BalanceSheetReportViewProps {
  clientId: string;
}

export default function BalanceSheetReportView({ clientId }: BalanceSheetReportViewProps) {
  const [projectionYears, setProjectionYears] = useState<ProjectionYear[]>([]);
  const [selectedYear, setSelectedYear] = useState<number | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function load() {
      const res = await fetch(`/api/clients/${clientId}/projection-data`);
      const data = await res.json();
      const years = runProjection(data);
      setProjectionYears(years);
      if (years.length > 0) setSelectedYear(years[0].year);
      setLoading(false);
    }
    load();
  }, [clientId]);

  if (loading) {
    return <div className="text-gray-400">Loading projection...</div>;
  }

  const yearData = projectionYears.find((y) => y.year === selectedYear);
  if (!yearData) return null;

  const categories = [
    { label: "Cash", key: "cash" as const, total: yearData.portfolioAssets.cashTotal },
    { label: "Taxable", key: "taxable" as const, total: yearData.portfolioAssets.taxableTotal },
    { label: "Retirement", key: "retirement" as const, total: yearData.portfolioAssets.retirementTotal },
    { label: "Real Estate", key: "realEstate" as const, total: yearData.portfolioAssets.realEstateTotal },
    { label: "Business", key: "business" as const, total: yearData.portfolioAssets.businessTotal },
    { label: "Life Insurance", key: "lifeInsurance" as const, total: yearData.portfolioAssets.lifeInsuranceTotal },
  ];

  const totalAssets = yearData.portfolioAssets.total;
  const totalLiabilities = yearData.expenses.liabilities; // This approximation may need refinement
  const netWorth = totalAssets; // Will refine once liabilities are tracked in portfolioAssets

  return (
    <div>
      <div className="mb-6 flex items-center justify-between">
        <h1 className="text-2xl font-bold text-gray-100">Balance Sheet</h1>
        <select
          value={selectedYear ?? ""}
          onChange={(e) => setSelectedYear(Number(e.target.value))}
          className="rounded border border-gray-700 bg-gray-900 px-3 py-1.5 text-sm text-gray-100"
        >
          {projectionYears.map((y) => (
            <option key={y.year} value={y.year}>
              {y.year}
            </option>
          ))}
        </select>
      </div>

      {/* Assets */}
      <div className="space-y-4">
        <h2 className="text-lg font-semibold text-gray-200">Assets</h2>
        {categories
          .filter((c) => c.total > 0)
          .map((cat) => (
            <div key={cat.key} className="rounded-lg border border-gray-700 bg-gray-900 p-4">
              <div className="flex items-center justify-between border-b border-gray-700 pb-2 mb-2">
                <span className="text-sm font-medium text-gray-300">{cat.label}</span>
                <span className="text-sm font-semibold text-gray-100">
                  ${cat.total.toLocaleString(undefined, { maximumFractionDigits: 0 })}
                </span>
              </div>
              {Object.entries(yearData.portfolioAssets[cat.key]).map(([name, value]) => (
                <div key={name} className="flex items-center justify-between py-1">
                  <span className="text-sm text-gray-400">{name}</span>
                  <span className="text-sm text-gray-200">
                    ${(value as number).toLocaleString(undefined, { maximumFractionDigits: 0 })}
                  </span>
                </div>
              ))}
            </div>
          ))}

        <div className="flex items-center justify-between rounded-lg bg-gray-800 px-4 py-3">
          <span className="font-semibold text-gray-200">Total Assets</span>
          <span className="font-semibold text-gray-100">
            ${totalAssets.toLocaleString(undefined, { maximumFractionDigits: 0 })}
          </span>
        </div>
      </div>

      {/* Net Worth */}
      <div className="mt-6 flex items-center justify-between rounded-lg bg-blue-900/30 border border-blue-800 px-4 py-3">
        <span className="font-bold text-gray-100">Net Worth</span>
        <span className="font-bold text-blue-300">
          ${netWorth.toLocaleString(undefined, { maximumFractionDigits: 0 })}
        </span>
      </div>
    </div>
  );
}
```
